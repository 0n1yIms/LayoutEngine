# CSS

ID -> ([a-z] | [A-Z]) ([a-z] | [A-Z] | [0-9] | '-' | '_')*

NUM -> ([0-9])+ (('.' [0-9]*) | ?)

HEX -> '#' ([a-f] | [A-F] | [0-9])+

SPACE -> (' ' | '\t' | '\n')*

LT -> '<'
GT -> '>'
SLASH -> '/'
EQUALS -> '='
QUOTE -> '"'
MULT -> '*'
BRAKET_O -> '{'
BRAKET_C -> '}'
DOT -> '.'
NUMERAL -> '#'
SEMICOLON -> ';'
COLON -> ':'
PERCENT -> '%'

COMMENT -> '/*' ([0-9] | [a-z] | [A-Z] | ' ' | '-')* '*/'
